        <h3>🛠️ أطوار اللعب</h3>
        <button class="menu-btn" onclick="startTargetMode('time')">⏱️ طور الوقت المحدد</button>
        <button class="menu-btn" onclick="startTargetMode('racers')">🏎️ طور المتسابقين الكثر</button>
        <button class="menu-btn" onclick="startTargetMode('obstacles')">⚠️ طور العوائق</button>

            </div>

                <span>🎨 لون ذهبي (50 🪙)</span>
                <button class="buy-btn" id="btn-gold" onclick="buyItem('gold', 50, 'color', '#ffd700', 'btn-gold')">شراء</button>

            <!-- الكفرات -->

                <span>🛞 كفرات رياضية (30 🪙)</span>

            </div>
            <!-- الهيكل -->
            <div class="shop-item">
                <span>🚀 هيكل ديناميكي (70 🪙)</span>

<script>
    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');

    // بيانات اللاعب والمشتريات (حفظ تلقائي)
    let playerData = JSON.parse(localStorage.getItem('carGameData')) || {
        coins: 100, // نقاط بدائية للتجربة
        unlocked: ['#1e90ff'],
        currentCustoms: { color: '#1e90ff', wheels: false, exhaust: false, spoiler: false }
    };

    // إعدادات اللعبة
    let gameMode = null; // 'time', 'racers', 'obstacles'
    let gameActive = false;
    let score = 0;
    let timer = 0;
    let gameInterval;
    
    // لاعب
    let car = { x: 180, y: 500, w: 40, h: 70, speed: 5 };
    // مصفوفات الكائنات
    let enemies = [];
    let obstacles = [];
    let coins = [];
    
    // التحكم بالأسهم
    let keys = {};
    window.addEventListener('keydown', e => keys[e.key] = true);
    window.addEventListener('keyup', e => keys[e.key] = false);

    function updateCoinsDisplay() {
        document.getElementById('coins-display').innerText = playerData.coins;
        saveData();
    }
    function saveData() { localStorage.setItem('carGameData', JSON.stringify(playerData)); }

    // تشغيل الأطوار
    function startTargetMode(mode) {
        gameMode = mode;
        gameActive = true;
        score = 0;
        enemies = [];
        obstacles = [];
        coins = [];
        car.x = 180;
        
        if(mode === 'time') {
            timer = 30; // 30 ثانية للتحدي
            document.getElementById('mode-status').innerText = `الوقت المتبقي: ${timer}ث`;
        } else if(mode === 'racers') {
            document.getElementById('mode-status').innerText = `تجاوز المتسابقين!`;
        } else if(mode === 'obstacles') {
            document.getElementById('mode-status').innerText = `تجنب العوائق الصخرية!`;
        }

        clearInterval(gameInterval);
        gameInterval = setInterval(gameLoop, 1000 / 60);

        // مؤشر الوقت الخاص بطور الوقت
        if(mode === 'time') {
            let timerId = setInterval(() => {
                if(!gameActive) { clearInterval(timerId); return; }
                timer--;
                document.getElementById('mode-status').innerText = `الوقت المتبقي: ${timer}ث`;
                if(timer <= 0) {
                    endGame(true, "انتهى الوقت! فزت بـ 20 نقطة إضافية.");
                    playerData.coins += 20;
                    updateCoinsDisplay();
                    clearInterval(timerId);
                }
            }, 1000);
        }
    }

    // المتجر والتطبيق
    function buyItem(id, cost, type, value, btnId) {
        if(playerData.unlocked.includes(id)) {
            applyCustom(type, value);
            return;
        }
        if(playerData.coins >= cost) {
            playerData.coins -= cost;
            playerData.unlocked.push(id);
            applyCustom(type, value);
            document.getElementById(btnId).innerText = "تطبيق";
            document.getElementById(btnId).classList.add('owned');
            updateCoinsDisplay();
        } else {
            alert("النقاط غير كافية!");
        }
    }

    function applyCustom(type, value) {
        playerData.currentCustoms[type] = value;
        saveData();
    }

    // تحديث ورسم العناصر
    function gameLoop() {
        if(!gameActive) return;
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        // رسم خطوط الطريق (تأثير الحركة)
        ctx.fillStyle = "#555";
        ctx.fillRect(0, 0, canvas.width, canvas.height);
        ctx.fillStyle = "#fff";
        for(let i=0; i<canvas.height; i+=40) {
            ctx.fillRect(195, (i + (score*2)) % canvas.height, 10, 20);
        }

        // حركة اللاعب
        if((keys['ArrowLeft'] || keys['a']) && car.x > 20) car.x -= car.speed;
        if((keys['ArrowRight'] || keys['d']) && car.x < canvas.width - car.w - 20) car.x += car.speed;

        // رسم سيارة اللاعب مع التعديلات
        drawCar(car.x, car.y, playerData.currentCustoms);

        // زيادة النقاط التلقائية أثناء البقاء حياً
        score++;

        // توليد الكائنات بناءً على الطور
        if(score % 100 === 0) {
            coins.push({x: Math.random()*(canvas.width-60)+20, y: -20, w: 20, h: 20});
        }

        if(gameMode === 'racers' && score % 60 === 0) {
            enemies.push({x: Math.random()*(canvas.width-60)+20, y: -80, w: 40, h: 70, speed: Math.random()*3 + 2, color: '#e74c3c'});
        }
        
        if(gameMode === 'obstacles' && score % 80 === 0) {
            obstacles.push({x: Math.random()*(canvas.width-60)+20, y: -40, w: 50, h: 30});
        }
        
        if(gameMode === 'time' && score % 90 === 0) {
            enemies.push({x: Math.random()*(canvas.width-60)+20, y: -80, w: 40, h: 70, speed: 2, color: '#9b59b6'});
        }

        // تحريك ورسم العملات النقدية 🪙
        for(let i = coins.length - 1; i >= 0; i--) {
            coins[i].y += 4;
            ctx.fillStyle = "#ffd700";
            ctx.beginPath();
            ctx.arc(coins[i].x + 10, coins[i].y + 10, 10, 0, Math.PI*2);
            ctx.fill();

            // اصطدام مع العملة
            if(checkCollision(car, coins[i])) {
                playerData.coins += 5;
                updateCoinsDisplay();
                coins.splice(i, 1);
            }
        }

        // تحريك ورسم المنافسين 🏎️
        for(let i = enemies.length - 1; i >= 0; i--) {
            enemies[i].y += enemies[i].speed;
            drawCar(enemies[i].x, enemies[i].y, {color: enemies[i].color});

            if(checkCollision(car, enemies[i])) {
                endGame(false, "اصطدمت بمتسابق آخر!");
            }
        }

        // تحريك ورسم العوائق ⚠️
        for(let i = obstacles.length - 1; i >= 0; i--) {
            obstacles[i].y += 5;
            ctx.fillStyle = "#7f8c8d"; // صخرة رمادية
            ctx.fillRect(obstacles[i].x, obstacles[i].y, obstacles[i].w, obstacles[i].h);

            if(checkCollision(car, obstacles[i])) {
                endGame(false, "اصطدمت بالعائق!");
            }
        }
    }

    // دالة رسم السيارة بالتعديلات المصممة
    function drawCar(x, y, customs) {
        // العوادم (إكسسوار خلفي)
        if(customs.exhaust) {
            ctx.fillStyle = "#ff4500";
            ctx.fillRect(x + 5, y + 68, 8, 10);
            ctx.fillRect(x + 27, y + 68, 8, 10);
        }

        // جسم السيارة الأساسي
        ctx.fillStyle = customs.color || '#1e90ff';
        ctx.fillRect(x, y, 40, 70);

        // الهيكل الخارجي / الجناح (Spoiler)
        if(customs.spoiler) {
            ctx.fillStyle = "#111";
            ctx.fillRect(x - 5, y + 60, 50, 8);
        }

        // الكفرات الإطارات
        ctx.fillStyle = customs.wheels ? "#fff" : "#111"; // بيضاء رياضية أو سوداء عادية
        ctx.fillRect(x - 3, y + 10, 5, 12);  // أمامي يسار
        ctx.fillRect(x + 38, y + 10, 5, 12); // أمامي يمين
        ctx.fillRect(x - 3, y + 50, 5, 12);  // خلفي يسار
        ctx.fillRect(x + 38, y + 50, 5, 12); // خلفي يمين

        // زجاج السيارة
        ctx.fillStyle = "#e0f7fa";
        ctx.fillRect(x + 5, y + 20, 30, 15);
    }

    function checkCollision(rect1, rect2) {
        return rect1.x < rect2.x + rect2.w &&
               rect1.x + rect1.w > rect2.x &&
               rect1.y < rect2.y + rect2.h &&
               rect1.y + rect1.h > rect2.y;
    }

    function endGame(isWin, message) {
        gameActive = false;
        clearInterval(gameInterval);
        document.getElementById('mode-status').innerText = isWin ? `🏆 ${message}` : `💥 ${message}`;
        updateCoinsDisplay();
    }

    // تحديث النقاط عند التشغيل الأول
    updateCoinsDisplay();

    // فحص المشتريات السابقة لتحديث الأزرار عند تنشيط الصفحة
    playerData.unlocked.forEach(id => {
        let btn = document.getElementById('btn-' + id);
        if(btn) {
            btn.innerText = "تطبيق";
            btn.classList.add('owned');
        }
    });
</script>
